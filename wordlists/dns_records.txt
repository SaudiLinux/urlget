# سجلات AAAA (تحويل النطاق إلى عنوان IPv6)

# سجلات MX (خوادم البريد الإلكتروني)
example.com,MX,10 mail.attacker.com
example.com,MX,20 mail2.attacker.com

# سجلات NS (خوادم الأسماء)
example.com,NS,ns1.attacker.com
example.com,NS,ns2.attacker.com

# سجلات TXT (معلومات نصية)
example.com,TXT,v=spf1 ip4:192.168.1.0/24 -all
_dmarc.example.com,TXT,v=DMARC1; p=none; rua=mailto:admin@attacker.com

# سجلات SOA (بداية السلطة)
example.com,SOA,ns1.attacker.com admin.attacker.com 2023010101 3600 1800 604800 86400
